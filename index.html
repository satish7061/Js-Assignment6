Qu 6. Write a program to find the winner of the game?

  var Game_Winner = (str) => 
  {
    let a=0,d=0;
    for (let i=0;i< str.length;i++){
      if (str[i]=="A")
        a++;
      else
      if (str[i]=="D")
         d++;
    }
    if (a>d)
      return "Aditya";
    else
    if (d>a)
      return "Danish";
    else

  }
